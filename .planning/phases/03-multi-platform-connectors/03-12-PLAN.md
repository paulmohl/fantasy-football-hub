**Extend `frontend/src/lib/api.ts`** — add X-Rate-Limited detection in the response interceptor:

In the existing `api.interceptors.response.use(success_handler, error_handler)`, extend the success handler:
```typescript
api.interceptors.response.use(
  (res) => {
    if (res.headers['x-rate-limited'] === 'true') {
      window.dispatchEvent(new CustomEvent('rate-limited', {
        detail: { platform: res.config.url?.includes('yahoo') ? 'Yahoo' : 'ESPN' }
      }))
    }
    return res
  },
  async (err) => {
    // ... existing 401 refresh logic unchanged
  }
)
```

Note: The rate-limited custom event listener must be wired in App.tsx or a global component to show the toast. Add a useEffect in App.tsx:

useEffect(() => {

  }
  window.addEventListener('rate-limited', handler as EventListener)
  return () => window.removeEventListener('rate-limited', handler as EventListener)
}, [])
```

import { test, expect } from '@playwright/test'

test.describe('Yahoo Connect Flow', () => {
  test('Yahoo option appears on platform selection', async ({ page }) => {
    await page.goto('/connect')
    await expect(page.getByText('Yahoo')).toBeVisible()
  })

  test('clicking Yahoo triggers server-side redirect', async ({ page }) => {
    await page.goto('/connect')
    const [request] = await Promise.all([
      page.waitForRequest(req => req.url().includes('/auth/yahoo')),
      page.getByText('Yahoo').click(),
    ])
    expect(request.url()).toContain('/auth/yahoo')
  })

  test('connect?platform=yahoo shows Yahoo connected state', async ({ page }) => {
    await page.goto('/connect?platform=yahoo')
    await expect(page.getByText(/Yahoo/i)).toBeVisible()
  })

  test('connect?reconnect=yahoo shows Yahoo reconnect option', async ({ page }) => {
    await page.goto('/connect?reconnect=yahoo')
    await expect(page.getByText('Yahoo')).toBeVisible()
  })
})
```

import { test, expect } from '@playwright/test'

test.describe('Rate Limit Toast', () => {
  test('X-Rate-Limited header triggers toast notification', async ({ page }) => {
    await page.goto('/team')
    await page.route('/api/v1/yahoo/**', route =>
      route.fulfill({
        status: 200,
        headers: { 'X-Rate-Limited': 'true' },
        body: JSON.stringify({ data: [] }),
      })
    )
    await page.evaluate(() =>
      window.dispatchEvent(new CustomEvent('rate-limited', { detail: { platform: 'Yahoo' } }))
    )
    await expect(page.getByText(/rate-limited|cached results/i)).toBeVisible({ timeout: 3000 })
  })
})
