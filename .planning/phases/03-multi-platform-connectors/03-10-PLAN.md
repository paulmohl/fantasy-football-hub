**Backend: extend `backend/workers/tasks.py`** — add check_platform_credentials task:

```python
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
from sqlalchemy.orm import sessionmaker
from sqlalchemy import select
import base64
import json

from app.models.credential import UserCredential
from app.models.user import User
from app.services.credential_service import CredentialService
from app.services.yahoo_client import YahooClient, YahooAuthExpired
from app.services.espn_client import ESPNClient, ESPNAuthExpired


async def check_platform_credentials(ctx: dict) -> None:
    """MP-06: Validate all stored platform credentials; mark unhealthy on auth failure.

    Scheduled every 6 hours. For Yahoo: attempts a lightweight API call with current token,
    proactively refreshes if token is near expiry. For ESPN: verifies cookies via get_league_settings.
    On any auth error, sets UserCredential.is_healthy=False.
    """
    engine = create_async_engine(settings.database_url)
    async_session = sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)
    cred_svc = CredentialService()

    async with async_session() as db:
        result = await db.execute(
            select(UserCredential, User)
            .join(User, UserCredential.user_id == User.id)
            .where(UserCredential.is_healthy == True)
        )
        rows = result.all()

    for cred_row, user_row in rows:
        try:
            decrypted = json.loads(cred_svc.decrypt(user_row, cred_row.credentials_encrypted))
        except Exception as exc:
            logger.error("credential.check.decrypt_failed", user_id=str(user_row.id), platform=cred_row.platform, error=str(exc))
            async with async_session() as db:
                async with db.begin():
                    await cred_svc.mark_unhealthy(user_row.id, cred_row.platform, db)
            continue

        is_healthy = True
        async with httpx.AsyncClient(timeout=10.0) as http:
            if cred_row.platform == "yahoo":
                try:
                    yahoo = YahooClient(http, decrypted.get("access_token", ""))
                    await yahoo.get_game_key()
                except YahooAuthExpired:
                    is_healthy = False
                except Exception as exc:
                    logger.warning("credential.check.yahoo_error", user_id=str(user_row.id), error=str(exc))
            elif cred_row.platform == "espn":
                is_public = decrypted.get("is_public", False)
                if not is_public:
                    espn = ESPNClient(http, swid=decrypted.get("swid"), espn_s2=decrypted.get("espn_s2"))
                    league_id = str(user_row.id)[:8]
                    try:
                        await espn.get_league_settings(decrypted.get("league_id", "1"), 2025)
                    except ESPNAuthExpired:
                        is_healthy = False
                    except Exception as exc:
                        logger.warning("credential.check.espn_error", user_id=str(user_row.id), error=str(exc))

        if not is_healthy:
            async with async_session() as db:
                async with db.begin():
                    await cred_svc.mark_unhealthy(user_row.id, cred_row.platform, db)
            logger.info("credential.marked_unhealthy", user_id=str(user_row.id), platform=cred_row.platform)

    await engine.dispose()
```

```typescript
import { Link } from 'react-router-dom'
import { useAuthStore } from '@/store/auth'

export function HealthBanner() {
  const unhealthyPlatforms = useAuthStore((s) => s.unhealthyPlatforms)

  if (unhealthyPlatforms.length === 0) return null

  return (
    <div
      role="alert"
      className="bg-amber-500 text-white text-sm font-medium px-4 py-2 flex items-center justify-between shrink-0"
    >
      <span>
        {unhealthyPlatforms.length === 1
          ? `Your ${capitalize(unhealthyPlatforms[0])} connection has expired.`
          : `Your ${unhealthyPlatforms.map(capitalize).join(' and ')} connections have expired.`}
      </span>
      <div className="flex gap-3 ml-4">
        {unhealthyPlatforms.map((platform) => (
          <Link
            key={platform}
            to={`/connect?reconnect=${platform}`}
            className="underline underline-offset-2 whitespace-nowrap"
          >
            Reconnect {capitalize(platform)}
          </Link>
        ))}
      </div>
    </div>
  )
}

function capitalize(s: string) {
  return s.charAt(0).toUpperCase() + s.slice(1)
}
```

```typescript
function RequireAuth({ children }: { children: React.ReactNode }) {
  const token = useAuthStore((s) => s.token)
  const setUnhealthyPlatforms = useAuthStore((s) => s.setUnhealthyPlatforms)

  useEffect(() => {
    if (!token) return
    api.get('/users/me').then(({ data }) => {
      const unhealthy = (data.credential_health ?? [])
        .filter((c: { platform: string; is_healthy: boolean }) => !c.is_healthy)
        .map((c: { platform: string }) => c.platform)
      setUnhealthyPlatforms(unhealthy)
    }).catch(() => {})
  }, [token])

  if (!token) return <Navigate to="/login" replace />
  return <>{children}</>
}
```
